**----This will ignore Null values----**
SELECT COUNT (ALL AREA) FROM DOCTOR;

**----This will include the Null values----**
SELECT COUNT (*) FROM DOCTOR;

**----Select distinct area----------**
SELECT COUNT(DISTINCT AREA) FROM DOCTOR;


SELECT COUNT(SALPERMON), SUM(SALPERMON),
AVG(SALPERMON), MIN(SALPERMON), MAX(SALPERMON) 
FROM DOCTOR;

**----GROUP BY FUNCTIONS--------------**
SELECT AREA, SUM(SALPERMON)
FROM DOCTOR
GROUP BY AREA;


SELECT AREA, DOC_NAME, SUM(SALPERMON)
FROM DOCTOR
GROUP BY AREA, DOC_NAME;


SELECT AREA, MAX(SALPERMON)
FROM DOCTOR
GROUP BY AREA;


SELECT AREA, SUM(SALPERMON)
FROM DOCTOR
GROUP BY AREA;

SELECT AREA, SUM(SALPERMON)
FROM DOCTOR
GROUP BY AREA
HAVING SUM(SALPERMON) > 30000;

SELECT AREA, SUM(SALPERMON)
FROM DOCTOR
GROUP BY AREA
HAVING SUM(SALPERMON) < 10000;
