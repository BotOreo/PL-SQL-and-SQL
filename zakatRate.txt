set serveroutput on;
set verify off;

create or replace function calcZakat (p_sal IN number)
return number is

	zakatrate number :=0.025;
	payzakat number;

begin 
	IF p_sal = 5000 THEN payzakat :=p_sal * zakatrate;
	ELSE payzakat:=0;
	END IF;

RETURN (payzakat);

end calcZakat;
/
