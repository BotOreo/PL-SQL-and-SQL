SELECT PT_LNAME, PT_FNAME, NEXTAPPTDATE, DOC_ID, DOC_NAME
FROM PATIENT
WHERE NEXTAPPTDATE > (SELECT NEXTAPPTDATE FROM PATIENT
WHERE PT_FNAME = 'Susan')
AND DOC_ID=(SELECT DOC_ID FROM PATIENT WHERE PT_FNAME='Susan');

SELECT DOC_NAME, D.AREA, TO_CHAR(SALPERMON, '$999,999')
"SALARY", TO_CHAR(high, '$999,999') "HIGHEST",
TO_CHAR(LOW, '$999,999') "LOWEST"
FROM DOCTOR D,
(SELECT AREA, MAX (SALPERMON) HIGH,
MIN (SALPERMON) LOW
FROM DOCTOR
WHERE AREA <> 'Director'
GROUP BY AREA) S
WHERE D.AREA = S.AREA
ORDER BY DOC_NAME;
