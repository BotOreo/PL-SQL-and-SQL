SELECT DOC_NAME, ANNUAL_BONUS
FROM DOCTOR
WHERE ANNUAL_BONUS > (SELECT ANNUAL_BONUS
FROM DOCTOR WHERE DOC_NAME='Lewis');

SELECT DOC_NAME, DATEHIRED
FROM DOCTOR
WHERE DATEHIRED > (SELECT DATEHIRED
FROM DOCTOR WHERE DOC_NAME='Thompson');


SELECT PT_FNAME, PT_LNAME, DOC_ID
FROM PATIENT
WHERE DOC_ID = (SELECT DOC_ID
FROM PATIENT WHERE PT_LNAME='James' AND PT_FNAME='Scott')
AND NEXTAPPTDATE > (SELECT NEXTAPPTDATE
FROM PATIENT WHERE PT_LNAME='Saillez');
